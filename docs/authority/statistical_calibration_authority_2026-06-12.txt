import numpy as np
from scipy.special import erf, logsumexp
from scipy.optimize import minimize

EPS = 1e-15

def norm_cdf(z):
    return 0.5 * (1.0 + erf(z / np.sqrt(2.0)))

def cell_probs(mu, sigma, lo, hi, loc_id, cluster_id, b_loc, log_k_cluster):
    """
    mu: shape [n]
    sigma: shape [n]
    lo, hi: shape [K], real-valued preimage boundaries; tails may be +/- np.inf
    loc_id: shape [n], int
    cluster_id: shape [n], int
    b_loc: shape [L]
    log_k_cluster: shape [C]
    returns probs: shape [n, K]
    """
    k = np.exp(log_k_cluster[cluster_id])
    center = mu + b_loc[loc_id]
    scale = k * sigma

    z_hi = (hi[None, :] - center[:, None]) / scale[:, None]
    z_lo = (lo[None, :] - center[:, None]) / scale[:, None]

    p = norm_cdf(z_hi) - norm_cdf(z_lo)
    p = np.clip(p, EPS, 1.0)
    p = p / p.sum(axis=1, keepdims=True)  # numerical only; should already sum to 1
    return p

def neg_loglik(theta, mu, sigma, y, lo, hi, loc_id, cluster_id, n_locs, n_clusters,
               prior_b0=None, prior_tau=None):
    b_loc = theta[:n_locs]
    log_k = theta[n_locs:n_locs + n_clusters]
    p = cell_probs(mu, sigma, lo, hi, loc_id, cluster_id, b_loc, log_k)
    nll = -np.sum(np.log(p[np.arange(len(y)), y]))

    # Optional Gaussian shrinkage prior for MAP; empirical Bayes section below gives prior.
    if prior_b0 is not None and prior_tau is not None:
        nll += 0.5 * np.sum(((b_loc - prior_b0) / prior_tau) ** 2)
        nll += np.sum(np.log(prior_tau))

    return nll

def fit_bias_scale(mu, sigma, y, lo, hi, loc_id, cluster_id):
    n_locs = int(loc_id.max()) + 1
    n_clusters = int(cluster_id.max()) + 1
    theta0 = np.r_[np.zeros(n_locs), np.zeros(n_clusters)]  # b=0, k=1
    res = minimize(
        neg_loglik, theta0,
        args=(mu, sigma, y, lo, hi, loc_id, cluster_id, n_locs, n_clusters),
        method="L-BFGS-B"
    )
    b_hat = res.x[:n_locs]
    k_hat = np.exp(res.x[n_locs:n_locs + n_clusters])
    return b_hat, k_hat, res

def distance_from_mode(q):
    # ordered bins; tails can be handled by assigning their natural end index
    n, K = q.shape
    mode = np.argmax(q, axis=1)
    idx = np.arange(K)[None, :]
    return np.abs(idx - mode[:, None])

def softmax_rows(eta):
    z = eta - eta.max(axis=1, keepdims=True)
    e = np.exp(z)
    return e / e.sum(axis=1, keepdims=True)

def calibrated_probs(q, dist_class, gamma, beta):
    eta = gamma * np.log(np.clip(q, EPS, 1.0)) + beta[dist_class]
    return softmax_rows(eta)

def neg_logloss_cal(theta, q, y, dist_class, n_classes, lam_gamma=1.0, lam_beta=1.0):
    gamma = theta[0]
    # beta[0] fixed at 0 for identifiability
    beta = np.r_[0.0, theta[1:]]
    p = calibrated_probs(q, dist_class, gamma, beta)
    loss = -np.sum(np.log(p[np.arange(len(y)), y]))

    # shrink toward identity: gamma=1, beta all equal
    loss += 0.5 * lam_gamma * (gamma - 1.0) ** 2
    if len(beta) >= 3:
        d2 = beta[2:] - 2 * beta[1:-1] + beta[:-2]
        loss += 0.5 * lam_beta * np.sum(d2 ** 2)
    return loss

def brier_alpha(q, m, y):
    n, K = q.shape
    onehot = np.zeros_like(q)
    onehot[np.arange(n), y] = 1.0

    eq = q - onehot
    em = m - onehot

    Sq = np.mean(np.sum(eq * eq, axis=1))
    Sm = np.mean(np.sum(em * em, axis=1))
    C = np.mean(np.sum(eq * em, axis=1))

    denom = Sq + Sm - 2.0 * C
    if denom <= 1e-15:
        return 0.0
    return float(np.clip((Sm - C) / denom, 0.0, 1.0))

def logloss_alpha(q, m, y):
    from scipy.optimize import minimize_scalar

    def obj(a):
        p_y = a * q[np.arange(len(y)), y] + (1.0 - a) * m[np.arange(len(y)), y]
        return -np.sum(np.log(np.clip(p_y, EPS, 1.0)))

    res = minimize_scalar(obj, bounds=(0.0, 1.0), method="bounded")
    return float(res.x), res

def binary_class_alpha(qk, mk, yk, weights=None):
    if weights is None:
        weights = np.ones_like(qk, dtype=float)
    weights = weights / weights.sum()

    eq = qk - yk
    em = mk - yk
    Sq = np.sum(weights * eq * eq)
    Sm = np.sum(weights * em * em)
    C = np.sum(weights * eq * em)
    denom = Sq + Sm - 2.0 * C
    if denom <= 1e-15:
        return 0.0
    return float(np.clip((Sm - C) / denom, 0.0, 1.0))

def fee(price, c):
    return c * price * (1.0 - price)

def quote_interval(yes_bid, yes_ask, no_bid, no_ask, c):
    yes_bid_net = yes_bid - fee(yes_bid, c)
    yes_ask_gross = yes_ask + fee(yes_ask, c)
    no_bid_net = no_bid - fee(no_bid, c)
    no_ask_gross = no_ask + fee(no_ask, c)

    lower = np.maximum(yes_bid_net, 1.0 - no_ask_gross)
    upper = np.minimum(yes_ask_gross, 1.0 - no_bid_net)
    return lower, upper

def simplex_normalize(x):
    x = np.clip(x, EPS, None)
    return x / x.sum(axis=-1, keepdims=True)

def market_raw_probs(yes_bid, yes_ask, no_bid, no_ask, c):
    lo, hi = quote_interval(yes_bid, yes_ask, no_bid, no_ask, c)
    raw = 0.5 * (lo + hi)
    return simplex_normalize(raw), lo, hi

def power_debias(raw, gamma):
    z = np.power(np.clip(raw, EPS, 1.0), gamma)
    return z / z.sum(axis=1, keepdims=True)

def fit_power_gamma(raw, y):
    from scipy.optimize import minimize_scalar

    def obj(log_gamma):
        gamma = np.exp(log_gamma)
        p = power_debias(raw, gamma)
        return -np.sum(np.log(np.clip(p[np.arange(len(y)), y], EPS, 1.0)))

    res = minimize_scalar(obj, bounds=(-3.0, 3.0), method="bounded")
    return float(np.exp(res.x)), res

def effective_n(weights, corr):
    w = np.asarray(weights, dtype=float)
    w = w / w.sum()
    return 1.0 / float(w @ corr @ w)

def family_effective_n(family_sizes, rho_within, rho_between=0.0):
    n = np.asarray(family_sizes, dtype=float)
    N = n.sum()
    within_pairs = np.sum(n * (n - 1.0))
    between_pairs = N * N - np.sum(n * n)
    denom = N + rho_within * within_pairs + rho_between * between_pairs
    return float(N * N / denom)

def kelly_fraction(theta_pay, cost):
    # fraction of bankroll spent as token cost
    raw = (theta_pay - cost) / (1.0 - cost)
    return np.maximum(0.0, raw)

def derived_bound_from_fractional_kelly(theta_samples, cost, lambda_frac):
    theta_bar = float(np.mean(theta_samples))
    if theta_bar <= cost:
        return theta_bar, 1.0, 0.0  # no trade

    theta_bound = cost + lambda_frac * (theta_bar - cost)
    gamma_star = float(np.mean(theta_samples <= theta_bound))
    f = float(kelly_fraction(theta_bound, cost))
    return theta_bound, gamma_star, f

def realized_return_per_cost(paid, cost):

def fit_fractional_kelly_lambda(f_kelly, realized_ret):
    from scipy.optimize import minimize_scalar

    # require 1 + lambda * f * X > 0
    def obj(lam):
        mult = 1.0 + lam * f_kelly * realized_ret
        if np.any(mult <= 0.0):
            return np.inf
        return -np.sum(np.log(mult))

    res = minimize_scalar(obj, bounds=(0.0, 1.0), method="bounded")
    return float(res.x), res

def approx_lambda(f_kelly, realized_ret):
    num = np.sum(f_kelly * realized_ret)
    den = np.sum((f_kelly * realized_ret) ** 2)
    if den <= 1e-15:
        return 0.0
    return float(np.clip(num / den, 0.0, 1.0))
