BLOCKER ‚ÄĒ Q1 / API design ‚ÄĒ proposed exit_policy.exit(...)

HIGH ‚ÄĒ Q1 / model risk ‚ÄĒ proposed calibration.blended_exit_belief

Issue: with a miscalibrated agent posterior, the pure Bayesian sell-dominance rule can systematically refuse exits because it trusts q_t while the market is informative.

Impact: the system holds losers whenever the model is stale, biased, or missing information already incorporated in the book.

Fix direction: fit a blended belief using the market as a second forecaster, and add an anytime-valid calibration alarm that suspends agent-posterior authority when out-of-sample likelihood evidence crosses a cost-derived threshold.

HIGH ‚ÄĒ Q2 / sizing ‚ÄĒ proposed portfolio_kelly.solve_joint_kelly

Issue: per-candidate sizing ignores mutual exclusivity within an event, positive same-day correlations across events, and bankroll lockup.

Impact: positions look individually small but can be collectively overbet on the same latent weather regime or event family.

Fix direction: exact horse-race Kelly inside each K-bin event; scenario/robust Kelly across event families; liquidity-reservation dynamic program for asynchronous opportunities.

HIGH ‚ÄĒ Q1/Q2/Q3 / liquidity ‚ÄĒ proposed execution.depth_aware_value

Issue: ‚Äúno price impact‚ÄĚ is explicitly unsafe at size. Bid/ask top-of-book prices are not executable for full desired size.

Impact: exit, entry, and quote decisions overstate log value.

Fix direction: every decision must consume a depth curve and compute marginal executable prices. The correct objective uses integrated sale/buy proceeds, not a scalar bid/ask.

MEDIUM ‚ÄĒ Q3 / re-quoting ‚ÄĒ proposed execution.dynamic_quote_policy

Issue: a one-shot maker/taker threshold ignores state changes, queue progress, and cancel-replace optionality.

Impact: stale resting orders can remain live after posterior/book jumps; conversely, useful queue priority can be thrown away by na√Įve chasing.

Fix direction: event-driven Bellman re-decision; queue-aware Erlang fill transforms; replay-based estimate of dynamic option value over one-shot.

MEDIUM ‚ÄĒ Q4 / constants ‚ÄĒ proposed ops_constants.fit_decision_constants

Issue: operational constants should not be fixed hours/counts/percentiles.

Impact: hidden overfitting and brittle behavior across market regimes.

Fix direction: convert each constant into a fitted stochastic rate or a cost-derived posterior decision threshold.

import numpy as np
from scipy.optimize import minimize, minimize_scalar, brentq
from scipy.special import expit, logit
from scipy.stats import norm

EPS = 1e-12

def _clip_prob(q, eps=1e-9):
    return float(np.clip(q, eps, 1.0 - eps))

def sell_all_dominance_gap(
    q,
    net_bid,
    position_units,
    wealth_ex_position,
    t_remaining_days=0.0,
    g_star=0.0,
    win_payout=1.0,
    lose_payout=0.0,
):
    """
    Positive gap means selling the entire position at net_bid strictly dominates
    holding to settlement. position_units is max payout units, not cost basis.
    """
    q = _clip_prob(q)
    n = float(position_units)
    W = float(wealth_ex_position)
    if n <= 0 or W <= 0:
        raise ValueError("position_units and wealth_ex_position must be positive")

    M = np.exp(g_star * t_remaining_days)
    w_sell = W + n * net_bid * M
    w_win = W + n * win_payout
    w_lose = W + n * lose_payout
    if min(w_sell, w_win, w_lose) <= 0:
        return -np.inf

    u_sell = np.log(w_sell)
    u_hold = q * np.log(w_win) + (1.0 - q) * np.log(w_lose)
    return u_sell - u_hold

def take_profit_net_bid_threshold(
    q,
    position_units,
    wealth_ex_position,
    t_remaining_days=0.0,
    g_star=0.0,
    win_payout=1.0,
    lose_payout=0.0,
):
    """
    Minimum net executable bid per contract for sell-all dominance.
    """
    q = _clip_prob(q)
    n = float(position_units)
    W = float(wealth_ex_position)
    z = n / W
    M = np.exp(g_star * t_remaining_days)
    if z <= 1e-12:
        return (q * win_payout + (1.0 - q) * lose_payout) / M
    hold_ce_gross = np.exp(
        q * np.log1p(z * win_payout)
        + (1.0 - q) * np.log1p(z * lose_payout)
    )
    return (hold_ce_gross - 1.0) / (z * M)

def exit_fraction_binary(
    q,
    bid,
    position_units,
    wealth_ex_position,
    t_remaining_days=0.0,
    g_star=0.0,
    sell_fee_per_contract=0.0,
    win_payout=1.0,
    lose_payout=0.0,
    q_sd=0.0,
    z_quantile=0.0,
    depth=None,
):
    """
    Returns fraction of current position to sell.

    depth: optional list [(qty, net_bid_before_fee), ...] sorted from best to worse.
           Quantities are contract units available at each level.
           If depth is None, bid - sell_fee_per_contract is assumed executable
           for the whole position. That assumption must be disabled in production
           when displayed depth is finite.
    """
    q_eff = _clip_prob(q - z_quantile * q_sd)
    n = float(position_units)
    W = float(wealth_ex_position)
    if n <= 0:
        return 0.0
    if W <= 0:
        raise ValueError("wealth_ex_position must be positive")

    Mopp = float(np.exp(g_star * t_remaining_days))

    if depth is None:
        net = max(0.0, float(bid) - float(sell_fee_per_contract))

        def sale_cash(x):
            return n * float(x) * net * Mopp

    else:
        levels = [(float(qty), float(px)) for qty, px in depth if qty > 0]

        def sale_cash(x):
            qty_to_sell = n * float(np.clip(x, 0.0, 1.0))
            rem = qty_to_sell
            cash = 0.0
            for qty, px in levels:
                take = min(rem, qty)
                if take <= 0:
                    break
                cash += take * max(0.0, px - sell_fee_per_contract)
                rem -= take
            # Do not assume hidden liquidity.
            if rem > 1e-9:
                return -np.inf
            return cash * Mopp

    def utility(x):
        C = sale_cash(x)
        if not np.isfinite(C):
            return -np.inf
        w1 = W + C + n * (1.0 - x) * win_payout
        w0 = W + C + n * (1.0 - x) * lose_payout
        if w1 <= 0 or w0 <= 0:
            return -np.inf
        return q_eff * np.log(w1) + (1.0 - q_eff) * np.log(w0)

    res = minimize_scalar(lambda x: -utility(x), bounds=(0.0, 1.0), method="bounded")
    candidates = [(0.0, utility(0.0)), (1.0, utility(1.0))]
    if res.success:
        candidates.append((float(res.x), utility(float(res.x))))
    return max(candidates, key=lambda t: t[1])[0]

def one_step_information_option_value(q, sigma_q, sale_log_value, hold_log_lose, hold_log_win):
    """
    Normal approximation to E[max(S, H(Q'))] - max(S, H(q)).
    Use Monte Carlo from posterior transition samples in production.
    """
    q = _clip_prob(q)
    sigma_q = max(float(sigma_q), 0.0)
    S = float(sale_log_value)
    L0 = float(hold_log_lose)
    L1 = float(hold_log_win)
    D = L1 - L0
    if sigma_q <= 0 or abs(D) <= 1e-12:
        return max(S, L0 + q * D) - max(S, L0 + q * D)

    q0 = (S - L0) / D
    d = (q - q0) / sigma_q
    expected_max = (
        S * norm.cdf(-d)
        + (L0 + D * q) * norm.cdf(d)
        + D * sigma_q * norm.pdf(d)
    )
    return expected_max - max(S, L0 + q * D)

def fit_blended_logistic_forecaster(y, q_agent, q_market, l2=1e-4):
    """
    Logistic stacking model:
        logit P(Y=1) = beta0 + beta_a logit(q_agent) + beta_m logit(q_market)

    Returns beta and approximate covariance from inverse Hessian.
    Add hierarchical effects locally if enough data by family/city/bin.
    """
    y = np.asarray(y, dtype=float)
    qa = np.clip(np.asarray(q_agent, dtype=float), 1e-6, 1 - 1e-6)
    qm = np.clip(np.asarray(q_market, dtype=float), 1e-6, 1 - 1e-6)
    X = np.column_stack([np.ones_like(y), logit(qa), logit(qm)])

    def nll(beta):
        eta = X @ beta
        # stable logistic negative log-likelihood with L2 penalty
        loss = np.sum(np.logaddexp(0.0, eta) - y * eta)
        loss += 0.5 * l2 * np.sum(beta[1:] ** 2)
        return loss

    def grad(beta):
        p = expit(X @ beta)
        g = X.T @ (p - y)
        g[1:] += l2 * beta[1:]
        return g

    res = minimize(nll, np.zeros(X.shape[1]), jac=grad, method="BFGS")
    if not res.success:
        raise RuntimeError(res.message)
    beta = res.x
    cov = np.asarray(res.hess_inv)
    return beta, cov

def predict_blended_q(q_agent, q_market, beta, beta_cov=None, n_mc=0, rng=None):
    qa = _clip_prob(q_agent)
    qm = _clip_prob(q_market)
    x = np.array([1.0, logit(qa), logit(qm)])
    if beta_cov is None or n_mc <= 0:
        return _clip_prob(expit(x @ beta))
    rng = np.random.default_rng(rng)
    draws = rng.multivariate_normal(np.asarray(beta), np.asarray(beta_cov), size=n_mc)
    return _clip_prob(np.mean(expit(draws @ x)))

import numpy as np
from scipy.optimize import minimize

def horse_race_kelly(q, p, tol=1e-10):
    """
    Exact Kelly fractions for a mutually exclusive K-outcome YES market.
    q: posterior probabilities, sums to 1.
    p: effective ask prices per $1 payout, including fees.
    Returns (f, cash).
    """
    q = np.asarray(q, dtype=float)
    p = np.asarray(p, dtype=float)
    q = q / q.sum()
    if np.any(p <= 0):
        raise ValueError("all prices must be positive")

    K = len(q)
    ratios = q / p

    # Underround or fair enough: no-cash all-active solution.
    if p.sum() <= 1.0 + tol:
        f = q.copy()
        return f, max(0.0, 1.0 - f.sum())

    order = np.argsort(-ratios)
    best = None

    # Scan active sets ordered by q/p.
    for m in range(K + 1):
        A = order[:m]
        Q_A = q[A].sum() if m else 0.0
        P_A = p[A].sum() if m else 0.0
        if P_A >= 1.0 - tol:
            continue

        s = (1.0 - Q_A) / (1.0 - P_A)
        if s < -tol or s > 1.0 + tol:
            continue

        min_active = ratios[A].min() if m else np.inf
        max_inactive = ratios[order[m:]].max() if m < K else -np.inf

        if min_active + tol >= s and s + tol >= max_inactive:
            f = np.maximum(q - p * s, 0.0)
            cash = 1.0 - f.sum()
            wealth = cash + f / p
            if np.any(wealth[q > tol] <= 0):
                obj = -np.inf
            else:
                obj = np.sum(q[q > tol] * np.log(wealth[q > tol]))
            if best is None or obj > best[0]:
                best = (obj, f, cash)

    if best is None:
        return np.zeros_like(q), 1.0

    return best[1], best[2]

    if payout is None:
        payout = np.ones(N)
    payout = np.asarray(payout, dtype=float)
    if corr is None:
        corr = np.eye(N)
    corr = np.asarray(corr, dtype=float)

    scale = payout / p
    mu = scale * q - 1.0

    EYij = np.outer(q, q)
    sd = np.sqrt(q * (1.0 - q))
    EYij += corr * np.outer(sd, sd)

    # Enforce exact diagonal and Frechet bounds.
    for i in range(N):
        EYij[i, i] = q[i]
    lower = np.maximum(0.0, q[:, None] + q[None, :] - 1.0)
    upper = np.minimum(q[:, None], q[None, :])
    EYij = np.minimum(np.maximum(EYij, lower), upper)

    M = np.outer(scale, scale) * EYij
    M -= np.outer(scale * q, np.ones(N))
    M -= np.outer(np.ones(N), scale * q)
    M += 1.0
    return mu, M

def kelly_qp(mu, M, payoff_states=None, budget=1.0, bounds=None, eps=1e-9):
    """
    Second-order Kelly approximation:
        maximize mu'f - 0.5 f'Mf
    with optional exact positivity constraints 1 + X_s f >= eps.
    """
    mu = np.asarray(mu, dtype=float)
    M = np.asarray(M, dtype=float)
    N = len(mu)
    if bounds is None:
        bounds = [(0.0, budget) for _ in range(N)]

    def obj(f):
        return -(mu @ f - 0.5 * f @ M @ f)

    def grad(f):
        return -(mu - M @ f)

    cons = [{"type": "ineq", "fun": lambda f: budget - np.sum(f)}]
    if payoff_states is not None:
        X = np.asarray(payoff_states, dtype=float)
        for s in range(X.shape[0]):
            cons.append({
                "type": "ineq",
                "fun": lambda f, xs=X[s]: 1.0 + xs @ f - eps
            })

    x0 = np.zeros(N)
    res = minimize(obj, x0, jac=grad, bounds=bounds, constraints=cons, method="SLSQP")
    if not res.success:
        raise RuntimeError(res.message)
    return res.x

def exact_scenario_kelly(payoff_states, probs, budget=1.0, bounds=None, eps=1e-9):
    """
    Exact finite-scenario Kelly:
        maximize sum_s p_s log(1 + X_s f).
    """
    X = np.asarray(payoff_states, dtype=float)
    probs = np.asarray(probs, dtype=float)
    probs = probs / probs.sum()
    N = X.shape[1]
    if bounds is None:
        bounds = [(0.0, budget) for _ in range(N)]

    def obj(f):
        w = 1.0 + X @ f
        if np.any(w <= eps):
            return 1e100
        return -np.sum(probs * np.log(w))

    def grad(f):
        w = 1.0 + X @ f
        return -(X.T @ (probs / w))

    cons = [{"type": "ineq", "fun": lambda f: budget - np.sum(f)}]
    for s in range(X.shape[0]):
        cons.append({
            "type": "ineq",
            "fun": lambda f, xs=X[s]: 1.0 + xs @ f - eps
        })

    res = minimize(obj, np.zeros(N), jac=grad, bounds=bounds, constraints=cons, method="SLSQP")
    if not res.success:
        raise RuntimeError(res.message)
    return res.x

def robust_kelly_from_probability_samples(payoff_states, prob_samples, budget=1.0, bounds=None, eps=1e-9):
    """
    Robust Kelly over posterior samples of scenario probabilities:
        maximize tau
        s.t. tau <= sum_s pi_b[s] log(1+X_s f) for all b.
    """
    X = np.asarray(payoff_states, dtype=float)
    P = np.asarray(prob_samples, dtype=float)
    P = P / P.sum(axis=1, keepdims=True)
    B, S = P.shape
    N = X.shape[1]
    if bounds is None:
        bounds_f = [(0.0, budget) for _ in range(N)]
    else:
        bounds_f = bounds
    bounds_all = bounds_f + [(-np.inf, np.inf)]

    def obj(y):
        return -y[-1]

    cons = [{"type": "ineq", "fun": lambda y: budget - np.sum(y[:N])}]

    for s in range(S):
        cons.append({
            "type": "ineq",
            "fun": lambda y, xs=X[s]: 1.0 + xs @ y[:N] - eps
        })

    for b in range(B):
        cons.append({
            "type": "ineq",
            "fun": lambda y, pb=P[b]: np.sum(pb * np.log(1.0 + X @ y[:N])) - y[-1]
        })

    y0 = np.zeros(N + 1)
    y0[-1] = 0.0
    res = minimize(obj, y0, bounds=bounds_all, constraints=cons, method="SLSQP")
    if not res.success:
        raise RuntimeError(res.message)
    return res.x[:N], res.x[-1]

def taylor_remainder_bound(payoff_states, f):
    X = np.asarray(payoff_states, dtype=float)
    z = X @ np.asarray(f, dtype=float)
    rho = np.max(np.abs(z))
    if rho >= 1.0:
        return np.inf
    return np.mean(np.abs(z) ** 3 / (3.0 * (1.0 - rho)))

import numpy as np
from scipy.special import gammainc
from scipy.optimize import minimize

def erlang_fill_transforms(shape, mu, lambda_edge, H):
    """
    shape: positive integer effective queue units to fill.
    mu: queue depletion rate.
    lambda_edge: exponential edge decay rate.
    H: horizon in same time units.
    Returns F_r(H), G_r(H).
    """
    r = int(max(1, shape))
    mu = float(mu)
    le = float(lambda_edge)
    H = float(H)
    if mu <= 0 or H <= 0:
        return 0.0, 0.0

    F = gammainc(r, mu * H)
    rate = mu + le
    G = (mu / rate) ** r * gammainc(r, rate * H)
    return float(F), float(G)

def queue_rest_value(shape, delta, H, e0, mu, lambda_edge, adverse=0.0):
    F, G = erlang_fill_transforms(shape, mu, lambda_edge, H)
    return F * delta - e0 * (1.0 - G) - adverse

def chase_hold_cross_decision(
    old_shape,
    old_delta,
    old_adverse,
    new_shape,
    new_delta,
    new_adverse,
    H,
    e0,
    mu,
    lambda_edge,
    cancel_replace_cost=0.0,
):
    v_old = queue_rest_value(old_shape, old_delta, H, e0, mu, lambda_edge, old_adverse)
    v_new = queue_rest_value(new_shape, new_delta, H, e0, mu, lambda_edge, new_adverse) - cancel_replace_cost
    v_cross = 0.0
    vals = {"hold_queue": v_old, "chase_touch": v_new, "cross": v_cross}
    return max(vals, key=vals.get), vals

def single_level_surplus(delta, H, e0, lambda0, kappa, adverse_fn):
    lam = lambda0 * np.exp(-kappa * delta)
    F = 1.0 - np.exp(-lam * H)
    return F * (e0 + delta - adverse_fn(delta))

def optimize_ladder_exact(fill_probs, returns_if_filled, max_sizes, total_size):
    """
    Exact independent Bernoulli-fill ladder optimizer for a small number of levels.
    For production, replace independent Bernoulli scenarios with historical joint fill scenarios.
    """
    fill_probs = np.asarray(fill_probs, dtype=float)
    ret = np.asarray(returns_if_filled, dtype=float)
    max_sizes = np.asarray(max_sizes, dtype=float)
    L = len(fill_probs)

    # Enumerate 2^L fill scenarios.
    scenarios = np.array(
        [[(mask >> j) & 1 for j in range(L)] for mask in range(1 << L)],
        dtype=float
    )
    probs = np.ones(1 << L)
    for j in range(L):
        probs *= np.where(scenarios[:, j] > 0, fill_probs[j], 1.0 - fill_probs[j])
    Z = scenarios * ret[None, :]

    def obj(u):
        w = 1.0 + Z @ u
        if np.any(w <= 1e-12):
            return 1e100
        return -np.sum(probs * np.log(w))

    def grad(u):
        w = 1.0 + Z @ u
        return -(Z.T @ (probs / w))

    bounds = [(0.0, float(m)) for m in max_sizes]
    cons = [{"type": "ineq", "fun": lambda u: total_size - np.sum(u)}]
    res = minimize(obj, np.zeros(L), jac=grad, bounds=bounds, constraints=cons, method="SLSQP")
    if not res.success:
        raise RuntimeError(res.message)
    return res.x, -res.fun

def event_trigger_certified(margin, lipschitz, state_delta):
    """
    Returns True if recomputation is required by the 2 L |ds| >= margin rule.
    """
    lhs = 2.0 * np.dot(np.asarray(lipschitz, dtype=float), np.abs(np.asarray(state_delta, dtype=float)))
    return lhs >= margin

import numpy as np
from scipy.optimize import brentq
from scipy.stats import norm

def expected_positive_part_normal(mu, sigma):
    sigma = max(float(sigma), 1e-12)
    z = mu / sigma
    return sigma * (norm.pdf(z) + z * norm.cdf(z))

def expected_negative_part_normal(mu, sigma):
    sigma = max(float(sigma), 1e-12)
    z = mu / sigma
    return sigma * (norm.pdf(z) - z * norm.cdf(-z))

def license_gain_normal(mu, sigma, c_plus=1.0, c_minus=1.0, c_impl=0.0):
    """
    Cost-derived license rule:
        c_plus E[Delta_+] > c_minus E[(-Delta)_+] + c_impl
    """
    ep = expected_positive_part_normal(mu, sigma)
    en = expected_negative_part_normal(mu, sigma)
    return c_plus * ep > c_minus * en + c_impl

def k_from_cost_ratio(c_plus=1.0, c_minus=1.0, c_impl_over_sigma=0.0):
    """
    Solves for k in mu > k sigma under the truncated-normal utility license.
    """
    def f(k):
        return (
            c_plus * (norm.pdf(k) + k * norm.cdf(k))
            - c_minus * (norm.pdf(k) - k * norm.cdf(-k))
            - c_impl_over_sigma
        )

    # Expand bracket if necessary.
    lo, hi = -10.0, 10.0
    if f(lo) > 0:
        return lo
    if f(hi) < 0:
        return hi
    return brentq(f, lo, hi)

def optimal_poll_interval(poll_cost, jump_intensity, stale_regret_rate):
    """
    Delta* = sqrt(2 C_p / (lambda_j r)).
    Units must be consistent.
    """
    poll_cost = float(poll_cost)
    jump_intensity = float(jump_intensity)
    stale_regret_rate = float(stale_regret_rate)
    if poll_cost <= 0:
        return 0.0
    if jump_intensity <= 0 or stale_regret_rate <= 0:
        return np.inf
    return np.sqrt(2.0 * poll_cost / (jump_intensity * stale_regret_rate))

def staleness_boundary(regret_fn, wait_cost_fn, lo=0.0, hi=72.0):
    """
    Finds first age where stale-data regret exceeds wait cost.
    regret_fn(age) and wait_cost_fn(age) return log-utility units.
    """
    def gap(a):
        return regret_fn(a) - wait_cost_fn(a)

    if gap(lo) >= 0:
        return lo
    if gap(hi) < 0:
        return np.inf
    return brentq(gap, lo, hi)

def lcb_fractional_equivalent_lambda(q_hat, p, sigma_q, z_alpha):
    """
    For one isolated binary bet:
        full Kelly on q_hat - z sigma equals lambda times full Kelly on q_hat.
    """
    edge = q_hat - p
    if edge <= 0:
        return 0.0
    return max(0.0, min(1.0, 1.0 - z_alpha * sigma_q / edge))
