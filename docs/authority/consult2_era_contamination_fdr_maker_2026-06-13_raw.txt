BLOCKER ‚ÄĒ Q3 / multiplicity policy / ‚ÄúBH over 0-or-1 p-values‚ÄĚ
The current BH procedure is mathematically vacuous: with p-values in {0,1}, every 0 passes and every 1 fails, so multiplicity correction has no operational effect. Worse, the mutually exclusive bin structure violates the positive-dependence condition under which ordinary BH is justified; BY would be valid under arbitrary dependence but is still solving the wrong problem for bankroll growth. Fix: replace BH with posterior expected-log-utility thresholding plus hierarchical, correlation-aware edge shrinkage.

HIGH ‚ÄĒ Q1 / era-mixed settled history / semantic drift
Full pooling across pipeline eras is not defensible unless an explicit era-interaction test and predictive check fail to find practically meaningful drift. The default should be EB partial pooling on transformed parameters, not newest-only and not naive all-history pooling. Fix: fit pooled/free-era/EB models, use LRT or score tests on fixed era effects, and use a boundary-aware or bootstrap test for the era-variance component.

HIGH ‚ÄĒ Q2 / provider de-bias and fusion weights
Sequential ‚Äúde-bias then inverse residual variance weight‚ÄĚ is consistent only in the asymptotic no-drift/no-confounding case. In current conditions, provider bias error inflates residual covariance and distorts fusion weights; omitted provider√ólocation/era structure can create persistent center bias. Fix: estimate provider, location, and era effects jointly, then derive GLS/BLUE fusion weights from the joint residual covariance.

MEDIUM-HIGH ‚ÄĒ Q4 / maker-vs-taker execution
Resting is not the conservative default for a real positive edge when fill and decay rates are uncertain. Under max-min reasoning, maker value can collapse to zero as fill intensity goes to zero or edge decay becomes fast; crossing locks the edge now. Fix: default to taker for robust positive-edge candidates; rest only when conservative posterior lower bounds on fill/decay still make maker value exceed taker value.

    stat = 2.0 * (loglik_free - loglik_pooled)
    df = (n_eras - 1) * n_params_per_era + extra_df
    return stat, df


def score_test(U_delta, I_dd, I_dp=None, I_pp_inv=None):
    """
    Efficient score test for era interactions.
    U_delta: score vector for era deviations at pooled fit.
    I_dd: information for era deviations.
    I_dp: cross information era x nuisance.
    I_pp_inv: inverse nuisance information.
    """
    if I_dp is not None and I_pp_inv is not None:
        I_eff = I_dd - I_dp @ I_pp_inv @ I_dp.T
    else:
        I_eff = I_dd
    return float(U_delta.T @ np.linalg.solve(I_eff, U_delta))


def eb_era_diag(theta_hat, V_diag, tau2_grid=None):
    """
    Diagonal empirical-Bayes normal-means shrinkage.

    theta_hat: shape (E, p), era MLEs on transformed scale.
    V_diag: shape (E, p), diagonal variances from inverse Hessians.
    Returns posterior means, posterior variances, EB grand mean, tau^2.
    """
    theta_hat = np.asarray(theta_hat, float)
    V_diag = np.asarray(V_diag, float)
    E, p = theta_hat.shape

    if tau2_grid is None:
        positive = V_diag[V_diag > 0]
        lo = max(np.min(positive) * 1e-4, 1e-12)
        hi = max(np.var(theta_hat, axis=0).max() * 100.0, np.max(positive) * 100.0, lo * 10)
        tau2_grid = np.r_[0.0, np.exp(np.linspace(np.log(lo), np.log(hi), 400))]

    mu = np.zeros(p)
    tau2 = np.zeros(p)
    post_mean = np.zeros_like(theta_hat)
    post_var = np.zeros_like(theta_hat)

    for j in range(p):
        y = theta_hat[:, j]
        s2 = np.maximum(V_diag[:, j], 1e-15)

        best = None
        for t2 in tau2_grid:
            v = s2 + t2
            w = 1.0 / v
            muj = np.sum(w * y) / np.sum(w)
            nll = 0.5 * np.sum(np.log(v) + (y - muj) ** 2 / v)
            if best is None or nll < best[0]:
                best = (nll, t2, muj)

        _, t2, muj = best
        mu[j] = muj
        tau2[j] = t2

        if t2 <= 1e-15:
            post_mean[:, j] = muj
            post_var[:, j] = 0.0
        else:
            prec = 1.0 / s2 + 1.0 / t2
            post_var[:, j] = 1.0 / prec
            post_mean[:, j] = post_var[:, j] * (y / s2 + muj / t2)

    return post_mean, post_var, mu, tau2


def eb_era_full_given_sigma(theta_hat, V_list, theta0, Sigma):
    """
    Full-matrix posterior means given EB hyperparameters.
    theta_hat: (E,p)
    V_list: list/array of E covariance matrices (p,p)
    theta0: (p,)
    Sigma: (p,p)
    """
    theta_hat = np.asarray(theta_hat, float)
    theta0 = np.asarray(theta0, float)
    Sigma = np.asarray(Sigma, float)
    E, p = theta_hat.shape

    out_mean = np.zeros_like(theta_hat)
    out_cov = np.zeros((E, p, p))
    Sinv = np.linalg.inv(Sigma)

    for e in range(E):
        V = np.asarray(V_list[e], float)
        Vinv = np.linalg.inv(V)
        post_cov = np.linalg.inv(Vinv + Sinv)
        post_mean = post_cov @ (Vinv @ theta_hat[e] + Sinv @ theta0)
        out_mean[e] = post_mean
        out_cov[e] = post_cov

    return out_mean, out_cov


def observed_probs_with_confusion(true_probs, confusion):
    """
    true_probs[c] = P(true category c)
    confusion[o,c] = P(observed o | true c)
    returns observed_probs[o]
    """
    true_probs = np.asarray(true_probs, float)
    confusion = np.asarray(confusion, float)
    return confusion @ true_probs


def step_change_time_decay_bias(delta, lam, n_new):
    """
    Bias of exponential time-decay estimator after a step change.
    Target is newest-era theta_1 = theta_0 + delta.
    """
    return -delta * (lam ** n_new)

import numpy as np

def minvar_weights(Sigma, ridge=1e-10):
    Sigma = np.asarray(Sigma, float)
    M = Sigma.shape[0]
    S = Sigma + ridge * np.eye(M)
    one = np.ones(M)
    a = np.linalg.solve(S, one)
    return a / (one @ a)


def distorted_diag_weights(v, eps_var):
    """
    v: intrinsic residual variances.
    eps_var: Var(stage-1 bias errors).
    """
    denom = np.asarray(v, float) + np.asarray(eps_var, float)
    inv = 1.0 / denom
    return inv / inv.sum()


def first_order_weight_distortion(Sigma, Ceps):
    Sigma = np.asarray(Sigma, float)
    Ceps = np.asarray(Ceps, float)
    M = Sigma.shape[0]
    one = np.ones(M)

    A = np.linalg.inv(Sigma)
    w = minvar_weights(Sigma)
    dw = -A @ Ceps @ w + w * (one @ A @ Ceps @ w)
    return w, dw


def center_rms_bias(w, Ceps):
    w = np.asarray(w, float)
    Ceps = np.asarray(Ceps, float)
    return float(np.sqrt(w @ Ceps @ w))


def ledoit_identity_shrink(S, rho):
    S = np.asarray(S, float)
    mu = np.trace(S) / S.shape[0]
    return (1.0 - rho) * S + rho * mu * np.eye(S.shape[0])


def fit_twoway_fixed_effects(d, provider, location):
    """
    Basic OLS two-way fixed effects for long-form residuals:
    d_im = intercept + provider FE + location FE + error.

    provider, location are integer-coded arrays aligned with d.
    Drops provider 0 and location 0 as references.
    """
    d = np.asarray(d, float)
    provider = np.asarray(provider, int)
    location = np.asarray(location, int)

    M = provider.max() + 1
    L = location.max() + 1
    n = len(d)

    cols = [np.ones(n)]
    for m in range(1, M):
        cols.append((provider == m).astype(float))
    for ell in range(1, L):
        cols.append((location == ell).astype(float))

    X = np.column_stack(cols)
    beta_hat, *_ = np.linalg.lstsq(X, d, rcond=None)
    resid = d - X @ beta_hat
    rank = np.linalg.matrix_rank(X)

    intercept = beta_hat[0]
    provider_effect = np.zeros(M)
    provider_effect[1:] = beta_hat[1:M]
    location_effect = np.zeros(L)
    location_effect[1:] = beta_hat[M:]

    return {
        "intercept": intercept,
        "provider_effect": provider_effect,
        "location_effect": location_effect,
        "residual": resid,
        "rank": rank,
        "df_resid": n - rank,
    }


def provider_cov_from_wide_residuals(R):
    """
    R: shape (n_rows, M), residuals with np.nan for missing providers.
    Pairwise covariance then nearest-PSD repair by eigenvalue clipping.
    """
    R = np.asarray(R, float)
    n, M = R.shape
    S = np.zeros((M, M))
    counts = np.zeros((M, M))

    for a in range(M):
        for b in range(M):
            mask = np.isfinite(R[:, a]) & np.isfinite(R[:, b])
            counts[a, b] = mask.sum()
            if counts[a, b] >= 2:
                xa = R[mask, a] - np.nanmean(R[mask, a])
                xb = R[mask, b] - np.nanmean(R[mask, b])
                S[a, b] = xa @ xb / (counts[a, b] - 1)
            else:
                S[a, b] = np.nan

    # Fill missing covariances with zero off-diagonal and median variance diagonal.
    diag_med = np.nanmedian(np.diag(S))
    for a in range(M):
        for b in range(M):
            if not np.isfinite(S[a, b]):
                S[a, b] = diag_med if a == b else 0.0

    S = 0.5 * (S + S.T)
    vals, vecs = np.linalg.eigh(S)
    vals = np.maximum(vals, 1e-10)
    S_psd = (vecs * vals) @ vecs.T
    return S_psd, counts

import math
import numpy as np

def normal_cdf(x):
    x = np.asarray(x, float)
    return 0.5 * (1.0 + np.vectorize(math.erf)(x / math.sqrt(2.0)))


def bh_rejections(pvals, q):
    p = np.asarray(pvals, float)
    m = len(p)
    order = np.argsort(p)
    ranked = p[order]
    thresh = q * np.arange(1, m + 1) / m
    ok = ranked <= thresh
    reject = np.zeros(m, dtype=bool)
    if np.any(ok):
        k = np.max(np.nonzero(ok)[0])
        reject[order[:k + 1]] = True
    return reject


def by_rejections(pvals, q):
    p = np.asarray(pvals, float)
    m = len(p)
    Hm = np.sum(1.0 / np.arange(1, m + 1))
    return bh_rejections(p, q / Hm)


def continuous_edge_pvalue(edge_hat, edge_se):
    z = np.asarray(edge_hat, float) / np.asarray(edge_se, float)
    return 1.0 - normal_cdf(z)


def local_false_sign_rate(edge_samples, threshold=0.0):
    edge_samples = np.asarray(edge_samples, float)
    return np.mean(edge_samples <= threshold, axis=0)


def effective_num_tests_from_cov(S):
    """
    Li-Ji-style effective test count from correlation eigenvalues.
    """
    S = np.asarray(S, float)
    d = np.sqrt(np.maximum(np.diag(S), 1e-15))
    R = S / np.outer(d, d)
    vals = np.linalg.eigvalsh(0.5 * (R + R.T))
    vals = np.maximum(vals, 0.0)
    return float((vals.sum() ** 2) / np.sum(vals ** 2))


def expected_max_standard_normal_approx(N_eff):
    N_eff = max(float(N_eff), 1.000001)
    a = math.sqrt(2.0 * math.log(N_eff))
    b = (math.log(math.log(N_eff)) + math.log(4.0 * math.pi)) / (2.0 * a)
    return a - b


def eb_shrink_edges(e_hat, S, R=None, tau2_grid=None):
    """
    Correlation-aware normal EB shrinkage:
      e_hat | e ~ N(e, S)
      e ~ N(mu*1, tau2*R)

    Returns posterior mean m, posterior covariance V, EB mu, tau2.
    """
    e_hat = np.asarray(e_hat, float)
    S = np.asarray(S, float)
    N = len(e_hat)
    one = np.ones(N)

    if R is None:
        R = np.eye(N)
    else:
        R = np.asarray(R, float)

    if tau2_grid is None:
        avg_noise = np.mean(np.diag(S))
        empirical = np.var(e_hat)
        hi = max(100.0 * avg_noise, 100.0 * empirical, 1e-8)
        tau2_grid = np.r_[0.0, np.exp(np.linspace(np.log(1e-12), np.log(hi), 500))]

    best = None
    for tau2 in tau2_grid:
        C = S + tau2 * R
        C = 0.5 * (C + C.T)
        sign, logdet = np.linalg.slogdet(C)
        if sign <= 0:
            continue
        Cinv_one = np.linalg.solve(C, one)
        mu = (one @ np.linalg.solve(C, e_hat)) / (one @ Cinv_one)
        resid = e_hat - mu * one
        nll = 0.5 * (logdet + resid @ np.linalg.solve(C, resid))
        if best is None or nll < best[0]:
            best = (nll, mu, tau2)

    _, mu, tau2 = best

    if tau2 <= 1e-15:
        m = mu * one
        V = np.zeros((N, N))
        return m, V, mu, tau2

    Sinv = np.linalg.inv(S)
    Rinv = np.linalg.inv(R)
    Prec = Sinv + (1.0 / tau2) * Rinv
    V = np.linalg.inv(Prec)
    m = V @ (Sinv @ e_hat + (1.0 / tau2) * Rinv @ (mu * one))
    return m, V, mu, tau2


def kelly_fraction_binary(q, price, f_cap=0.05):
    """
    Fraction of bankroll spent on contract, no fees.
    """
    q = np.asarray(q, float)
    price = np.asarray(price, float)
    f = (q - price) / np.maximum(1.0 - price, 1e-15)
    return np.clip(f, 0.0, f_cap)


def expected_log_growth_binary(q, price, f):
    """
    Spend fraction f. If win, profit multiple on spent capital is (1-price)/price.
    If lose, spent capital is lost.
    """
    b = (1.0 - price) / np.maximum(price, 1e-15)
    return q * np.log1p(f * b) + (1.0 - q) * np.log1p(-f)


def select_by_posterior_utility(q_samples, prices, f_cap=0.05, min_prob_edge=0.90):
    """
    q_samples: shape (S, N), posterior calibrated probabilities.
    prices: shape (N,), contract ask prices for the chosen side.
    """
    q_samples = np.asarray(q_samples, float)
    prices = np.asarray(prices, float)
    q_bar = q_samples.mean(axis=0)

    f = kelly_fraction_binary(q_bar, prices, f_cap=f_cap)
    growth = expected_log_growth_binary(q_bar, prices, f)
    prob_edge = np.mean(q_samples > prices[None, :], axis=0)

    allowed = (growth > 0.0) & (prob_edge >= min_prob_edge)
    if not np.any(allowed):
        return None

    idx = np.argmax(np.where(allowed, growth, -np.inf))
    return {
        "idx": int(idx),
        "growth": float(growth[idx]),
        "fraction": float(f[idx]),
        "prob_edge": float(prob_edge[idx]),
        "q_mean": float(q_bar[idx]),
        "price": float(prices[idx]),
    }

import numpy as np

def maker_value_conservative(e_cross, price_improvement, lam_f, lam_e, D):
    """
    Conservative model: entire resting edge decays.
    e_rest0 = e_cross + price_improvement.
    """
    e_rest0 = e_cross + price_improvement
    rate = lam_f + lam_e
    if rate <= 0:
        return e_rest0 if D > 0 else 0.0
    return e_rest0 * lam_f / rate * (1.0 - np.exp(-rate * D))


def maker_value_with_persistent_improvement(e_cross, price_improvement, lam_f, lam_e, D):
    """
    Optimistic model: price improvement persists; only crossed edge decays.
    Use only after markout diagnostics support non-adverse passive fills.
    """
    fill_prob = 1.0 - np.exp(-lam_f * D)
    rate = lam_f + lam_e
    decayed_component = e_cross * lam_f / rate * (1.0 - np.exp(-rate * D))
    return price_improvement * fill_prob + decayed_component


def maker_taker_decision(e_cross_L, e_rest_improvement_L, lam_f_L, lam_e_U, D):
    """
    Max-min conservative decision.
    """
    if e_cross_L <= 0.0:
        return "NO_TRADE", 0.0, e_cross_L

    vm_L = maker_value_conservative(
        e_cross=e_cross_L,
        price_improvement=e_rest_improvement_L,
        lam_f=lam_f_L,
        lam_e=lam_e_U,
        D=D,
    )
    vt_L = e_cross_L

    if vm_L > vt_L:
        return "MAKE", vm_L, vt_L
    return "TAKE", vm_L, vt_L


def fit_edge_decay_grid(g_now, g_next, dt, lam_grid=None):
    """
    Fits g_next ~= exp(-lambda * dt) * g_now by grid SSE.
    Supports irregular dt.
    """
    g_now = np.asarray(g_now, float)
    g_next = np.asarray(g_next, float)
    dt = np.asarray(dt, float)

    mask = np.isfinite(g_now) & np.isfinite(g_next) & np.isfinite(dt) & (dt > 0)
    g_now, g_next, dt = g_now[mask], g_next[mask], dt[mask]

    if lam_grid is None:
        lam_grid = np.exp(np.linspace(np.log(1e-4), np.log(100.0), 1000))

    best = None
    for lam in lam_grid:
        pred = np.exp(-lam * dt) * g_now
        sse = np.sum((g_next - pred) ** 2)
        if best is None or sse < best[0]:
            best = (sse, lam)

    return {"lambda_e": best[1], "sse": best[0], "half_life": np.log(2.0) / best[1]}


def gamma_fill_update(a0, b0, filled, exposure_time):
    """
    Gamma prior/posterior for exponential fill hazard.
    Rate parameterization: density proportional to lambda^(a-1) exp(-b lambda).
    """
    filled = np.asarray(filled, int)
    exposure_time = np.asarray(exposure_time, float)
    return a0 + filled.sum(), b0 + exposure_time.sum()


def gamma_posterior_fill_prob(a, b, D):
    """
    E[1 - exp(-lambda D)] for lambda ~ Gamma(a,b).
    """
    return 1.0 - (b / (b + D)) ** a


def maker_value_mc(e_cross_samples, improvement_samples, lam_f_samples, lam_e_samples, D):
    e_cross_samples = np.asarray(e_cross_samples, float)
    improvement_samples = np.asarray(improvement_samples, float)
    lam_f_samples = np.asarray(lam_f_samples, float)
    lam_e_samples = np.asarray(lam_e_samples, float)

    vals = maker_value_conservative(
        e_cross=e_cross_samples,
        price_improvement=improvement_samples,
        lam_f=lam_f_samples,
        lam_e=lam_e_samples,
        D=D,
    )
    return {
        "mean": float(np.mean(vals)),
        "p05": float(np.quantile(vals, 0.05)),
        "p50": float(np.quantile(vals, 0.50)),
        "p95": float(np.quantile(vals, 0.95)),
    }
